# Card

<Description text="Exibe um cartão com cabeçalho, conteúdo e rodapé." />

<SelectVersion />

<PreviewComponent name="card" />

## Instalação

Instale o componente a partir da linha de comando.

```bash
yarn add @rempi-ui/card
```

## Uso

Importe o <strong>Card</strong> e junte todas as partes do componente.

```jsx
import { Button } from "@rempi-ui/button";
import { Card } from "@rempi-ui/card";
import { Flex } from "@rempi-ui/flex";
import { Switch } from "@rempi-ui/switch";
import { Typography } from "@rempi-ui/typography";
import { BellIcon, Check } from "lucide-react";

() => (
  <Card.Root>
    <Card.Header>
      <Card.Title>Notificações</Card.Title>
      <Card.Description>Você possuí 3 mensagens não lidas.</Card.Description>
    </Card.Header>
    <Card.Content flexDirection="column">
      <Flex alignItems="center">
        <BellIcon size={16} />
        <Flex flexDirection="column">
          <Typography
            fontSize="sm"
            fontWeight="semi-bold"
            color="high-contrast"
            lineHeight={1}
          >
            Notificações via push
          </Typography>
          <Typography
            fontSize="sm"
            fontWeight="regular"
            color="low-contrast"
            lineHeight={1}
          >
            Envie notificações para o dispositivo.
          </Typography>
        </Flex>
        <Switch.Thumb />
      </Flex>
      <div>
        {[
          {
            title: "Sua chamada foi confirmada.",
            description: "1 hora atrás",
          },
          {
            title: "Você possuí uma nova mensagem!",
            description: "1 hora atrás",
          },
          {
            title: "Sua inscrição irá expirar em breve!",
            description: "2 horas atrás",
          },
        ].map((notification, index, ar) => (
          <Flex key={index} flexDirection="row" alignItems="start">
            <Typography />
            <Flex flexDirection="column">
              <Typography
                fontSize="sm"
                fontWeight="semi-bold"
                color="high-contrast"
                lineHeight={1}
              >
                {notification.title}
              </Typography>
              <Typography
                fontSize="sm"
                fontWeight="regular"
                color="low-contrast"
                lineHeight={1}
              >
                {notification.description}
              </Typography>
            </Flex>
          </Flex>
        ))}
      </div>
    </Card.Content>
    <Card.Footer>
      <Button>
        <Check size={16} />
        Marcar todas como lida
      </Button>
    </Card.Footer>
  </Card.Root>
);
```
